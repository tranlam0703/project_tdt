cau 2

B = [2 5 2 -20; 10 -2 -1 3; 1 1 10 2; 1 -10 -2 3];
g = [20 20 30 20];
X = B/g;
disp('cau 2')
disp(X);
[M,N] = size(B);
if M ~= N
    disp('k phai ma tran vuong');
else
for m = 1:M
    row = abs(B(m,:) );
    d = sum(row) - row(m);
    if row(m) <= d
        disp('khong cheo troi');
    else
        disp('co cheo troi');
    end
end
end
