NHOM 05 - GIAI TICH SO 1

cau 4 pp Lagrange

syms x;
xi = [ 50 52 54 56];
y = [ 3.684 3.732 3.779 3.825];
n = length(x);
L = 0;
for i = 1:n
    P(i,1) = y(i);
end
for i = 1:n
    P = 1;
    for j = 1:n
        if i ~= j
            P = P * ((x-xi(j))/(xi(i)-xi(j)));
        end
    end
    L = L + y(i)*P;
end
L

   3.684000000000000

cau 4 PP Newton

syms x;
xi = [ 50 52 54 56];
y = [ 3.684 3.732 3.779 3.825];
xp = 55;
n = length(x);
for i = 1:n
    DD(i,1) = y(i);
end

for i = 2:n
    for j = 2:i
        DD(i,j) = (DD(i,j-1) - DD(i-1,j-1))/(x(i)-x(i-j+1));
    end
end
f = DD(1,1);
for i=n-1:-1:1
     f = f*(xp-x(i)) + DD(i,i);
end
DD
